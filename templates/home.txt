<!-- ===== REVIEW SLIDESHOW SCRIPT ===== -->

<script>

let reviews = []

let index = 0


fetch("/reviews_data")

.then(res => res.json())

.then(data => {

reviews = data.map(r => {

let stars = "⭐".repeat(r.stars)

return `
<b>${r.email}</b><br>
${r.room}<br>
${stars}<br>
"${r.message}"<br>
<small>${r.date}</small>
`

})

showReview()

setInterval(showReview, 3000)

})


function showReview(){

if(reviews.length === 0) return


document.getElementById("reviewBox").style.opacity = 0;

setTimeout(() => {

document.getElementById("reviewBox").innerHTML =
reviews[index];

document.getElementById("reviewBox").style.opacity = 1;

}, 300);

reviews[index]


index++


if(index >= reviews.length)

index = 0

}


function showReviewForm(){

document.getElementById("reviewForm")

.style.display = "block"

}

</script>
